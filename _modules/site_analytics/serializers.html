[docs]    def to_internal_value(self, data):
        new_data = {}
        for key in self.STATIC_FIELDS:
            try:
                new_data[key] = data.pop(key)
            except KeyError:
                pass
        new_data.update(data=data)
        return super().to_internal_value(new_data)
